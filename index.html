import numpy as np
import matplotlib.pyplot as plt
from sklearn.cluster import DBSCAN
from sklearn.datasets import make_moons

# Generate sample data (you can replace this with your dataset)
X, _ = make_moons(n_samples=300, noise=0.05, random_state=42)

# Plot the generated data
plt.figure(figsize=(8, 6))
plt.scatter(X[:, 0], X[:, 1], s=30)
plt.title('Generated Data')
plt.xlabel('Feature 1')
plt.ylabel('Feature 2')
plt.grid()
plt.show()

# DBSCAN clustering
dbscan = DBSCAN(eps=0.2, min_samples=5)  # You can adjust eps and min_samples based on your data
labels = dbscan.fit_predict(X)

# Create a scatter plot of the clusters
plt.figure(figsize=(8, 6))
unique_labels = set(labels)
colors = plt.cm.Spectral(np.linspace(0, 1, len(unique_labels)))

for k, col in zip(unique_labels, colors):
    if k == -1:
        # Noise points are colored black
        col = 'k'
        
    class_member_mask = (labels == k)
    xy = X[class_member_mask]
    plt.scatter(xy[:, 0], xy[:, 1], s=30, color=col, label=f'Cluster {k}' if k != -1 else 'Noise')

plt.title('DBSCAN Clustering')
plt.xlabel('Feature 1')
plt.ylabel('Feature 2')
plt.legend()
plt.grid()
plt.show()

# Output the clusters
for cluster_id in unique_labels:
    cluster_points = X[labels == cluster_id]
    print(f'Cluster {cluster_id}: {cluster_points}')

# Import necessary libraries
import numpy as np
import matplotlib.pyplot as plt
from graph_based_clustering import ConnectedComponentsClustering, SpanTreeConnectedComponentsClustering

# Generate synthetic data for clustering
X = np.random.rand(100, 2)  # 100 samples with 2 features

# Connected Components Clustering
connected_clustering = ConnectedComponentsClustering(threshold=0.3, metric="euclidean", n_jobs=-1)
labels_connected = connected_clustering.fit_predict(X)

# Minimum Spanning Tree Clustering
mst_clustering = SpanTreeConnectedComponentsClustering(n_clusters=5, metric="euclidean", n_jobs=-1)
labels_mst = mst_clustering.fit_predict(X)

# Plotting the results
plt.figure(figsize=(14, 6))

# Plot Connected Components Clustering
plt.subplot(1, 2, 1)
plt.scatter(X[:, 0], X[:, 1], c=labels_connected, cmap='viridis', s=50)
plt.title('Connected Components Clustering')
plt.xlabel('Feature 1')
plt.ylabel('Feature 2')

# Plot Minimum Spanning Tree Clustering
plt.subplot(1, 2, 2)
plt.scatter(X[:, 0], X[:, 1], c=labels_mst, cmap='viridis', s=50)
plt.title('Minimum Spanning Tree Clustering')
plt.xlabel('Feature 1')
plt.ylabel('Feature 2')

plt.tight_layout()
